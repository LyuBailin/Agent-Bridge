class AppError extends Error {
  constructor(message, statusCode = 500) {
    super(message);
    this.statusCode = statusCode;
    this.isOperational = true;
    Error.captureStackTrace(this, this.constructor);
  }
}

function handleAppError(error, res) {
  if (error.isOperational) {
    return res.status(error.statusCode).json({
      success: false,
      message: error.message
    });
  }
  
  console.error('Unhandled Error:', error);
  return res.status(500).json({
    success: false,
    message: 'Internal server error'
  });
}

module.exports = {
  AppError,
  handleAppError
};

const colors = {
  reset: '\x1b[0m',
  red: '\x1b[31m',
  green: '\x1b[32m',
  yellow: '\x1b[33m',
  blue: '\x1b[34m',
  magenta: '\x1b[35m',
  cyan: '\x1b[36m'
};

const levels = {
  debug: { color: colors.cyan, prefix: 'DEBUG' },
  info: { color: colors.green, prefix: 'INFO' },
  warn: { color: colors.yellow, prefix: 'WARN' },
  error: { color: colors.red, prefix: 'ERROR' }
};

function formatMessage(level, message, meta = {}) {
  const timestamp = new Date().toISOString();
  const levelConfig = levels[level] || levels.info;
  const metaStr = Object.keys(meta).length > 0 ? JSON.stringify(meta) : '';
  return `${levelConfig.color}[${timestamp}] [${levelConfig.prefix}]${colors.reset} ${message}${metaStr ? ' ' + metaStr : ''}`;
}

function log(level, message, meta = {}) {
  const formatted = formatMessage(level, message, meta);
  switch (level) {
    case 'error':
      console.error(formatted);
      break;
    default:
      console.log(formatted);
  }
}

const logger = {
  debug: (message, meta) => log('debug', message, meta),
  info: (message, meta) => log('info', message, meta),
  warn: (message, meta) => log('warn', message, meta),
  error: (message, meta) => log('error', message, meta)
};

module.exports = logger;
