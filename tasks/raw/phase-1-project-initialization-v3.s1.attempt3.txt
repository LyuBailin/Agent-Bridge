## Setup Instructions

### Prerequisites
- Node.js >= 18.x
- PostgreSQL >= 14.x
- Git

### Installation

1. Clone the repository
2. Install dependencies:
   ```bash
   npm install
   ```
3. Configure environment variables in `.env`
4. Initialize database:
   ```bash
   npm run db:init
   ```
5. Start development servers:
   ```bash
   npm run dev
   ```

## Database Schema

- **users**: User authentication and core account data
- **profiles**: Extended user profile information
- **data**: Application-specific data storage

## Security Notes

- Never commit `.env` files to version control
- Use environment variables for sensitive configuration
- Implement proper input validation and sanitization

  process.exit(-1);
});

module.exports = {
  query: (text, params) => pool.query(text, params),

-- PostgreSQL Database Schema for Phase 1 Project

-- Users table: Core user authentication and account data
CREATE TABLE IF NOT EXISTS users (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Profiles table: Extended user profile information
CREATE TABLE IF NOT EXISTS profiles (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    first_name VARCHAR(100),
    last_name VARCHAR(100),
    bio TEXT,
    avatar_url VARCHAR(500),
    phone VARCHAR(20),
    date_of_birth DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user_id)
);

-- Data table: Application-specific data storage
CREATE TABLE IF NOT EXISTS data (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    key VARCHAR(255) NOT NULL,
    value JSONB,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user_id, key)
);

-- Indexes for better query performance
CREATE INDEX IF NOT EXISTS idx_users_email ON users(email);
CREATE INDEX IF NOT EXISTS idx_users_username ON users(username);
CREATE INDEX IF NOT EXISTS idx_profiles_user_id ON profiles(user_id);
CREATE INDEX IF NOT EXISTS idx_data_user_id ON data(user_id);
CREATE INDEX IF NOT EXISTS idx_data_key ON data(key);

-- Trigger to update updated_at timestamp
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_users_updated_at BEFORE UPDATE ON users
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_profiles_updated_at BEFORE UPDATE ON profiles
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_data_updated_at BEFORE UPDATE ON data
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

const fs = require('fs');
const path = require('path');
const db = require('../config/database');

async function initializeDatabase() {
  try {
    const client = await db.getClient();
    
    console.log('Reading schema file...');
    const schemaPath = path.join(__dirname, 'schema.sql');
    const schema = fs.readFileSync(schemaPath, 'utf8');
    
    console.log('Creating database tables...');
    await client.query(schema);
    
    console.log('Database initialized successfully!');
    
    await client.release();
  } catch (error) {
    console.error('Database initialization failed:', error.message);
    throw error;
  }
}

initializeDatabase();

const db = require('../config/database');

class User {
  static async create(data) {
    const { email, username, password_hash } = data;
    const query = `
      INSERT INTO users (email, username, password_hash)
      VALUES ($1, $2, $3)
      RETURNING id, email, username, created_at
    `;
    const result = await db.query(query, [email, username, password_hash]);
    return result.rows[0];
  }

  static async findById(id) {
    const query = 'SELECT * FROM users WHERE id = $1';
    const result = await db.query(query, [id]);
    return result.rows[0] || null;
  }

  static async findByEmail(email) {
    const query = 'SELECT * FROM users WHERE email = $1';
    const result = await db.query(query, [email]);
    return result.rows[0] || null;
  }

  static async findByUsername(username) {
    const query = 'SELECT * FROM users WHERE username = $1';
    const result = await db.query(query, [username]);
    return result.rows[0] || null;
  }

  static async update(id, data) {
    const fields = Object.keys(data).filter(k => k !== 'id');
    const values = Object.values(data);
    const setClause = fields.map((f, i) => `${f} = $${i + 1}`).join(', ');
    const query = `
      UPDATE users SET ${setClause}, updated_at = CURRENT_TIMESTAMP
      WHERE id = $${fields.length + 1}
      RETURNING *
    `;
    const result = await db.query(query, [...values, id]);
    return result.rows[0] || null;
  }

  static async delete(id) {
    const query = 'DELETE FROM users WHERE id = $1 RETURNING id';
    const result = await db.query(query, [id]);
    return result.rowCount > 0;
  }
}

module.exports = User;

const db = require('../config/database');

class Data {
  static async create(data) {
    const { user_id, key, value, metadata } = data;
    const query = `
      INSERT INTO data (user_id, key, value, metadata)
      VALUES ($1, $2, $3, $4)
      RETURNING *
    `;
    const result = await db.query(query, [user_id, key, JSON.stringify(value), JSON.stringify(metadata)]);
    return result.rows[0];
  }

  static async findByUserAndKey(userId, key) {
    const query = 'SELECT * FROM data WHERE user_id = $1 AND key = $2';
    const result = await db.query(query, [userId, key]);
    return result.rows[0] || null;
  }

  static async update(userId, key, data) {
    const { value, metadata } = data;
    const query = `
      UPDATE data 
      SET value = $1, metadata = $2, updated_at = CURRENT_TIMESTAMP
      WHERE user_id = $3 AND key = $4
      RETURNING *
    `;
    const result = await db.query(query, [
      JSON.stringify(value),
      JSON.stringify(metadata),
      userId,
      key
    ]);
    return result.rows[0] || null;
  }

  static async delete(userId, key) {
    const query = 'DELETE FROM data WHERE user_id = $1 AND key = $2 RETURNING id';
    const result = await db.query(query, [userId, key]);
    return result.rowCount > 0;
  }

  static async findAllByUser(userId) {
    const query = 'SELECT * FROM data WHERE user_id = $1 ORDER BY created_at DESC';
    const result = await db.query(query, [userId]);
    return result.rows.map(row => ({
      ...row,
      value: row.value ? JSON.parse(row.value) : null,
      metadata: row.metadata ? JSON.parse(row.metadata) : null
    }));
  }
}

module.exports = Data;
